# 算法偏见：识别并缓解智能体决策中的偏见

## 引言

"AI是中立的"——这是一个危险的误解。AI Agent的决策并非从真空中产生，它们从带有偏见的数据中学习，使用由有偏见的人类设计的算法，服务于可能有偏见的任务目标。如果说传统软件中的Bug会导致功能错误，那么AI Agent中的**偏见（Bias）**则可能导致系统性的不公平——某些群体被系统性地忽视、歧视或伤害。

在多智能体系统中，偏见的危害会进一步放大：一个Agent的偏见可能通过协作传播到其他Agent，形成"偏见传播链"。更糟糕的是，Agent的决策往往看起来"客观"和"数据驱动"，这使得偏见更隐蔽、更难被发现。本章将系统分析Agent偏见问题的来源、检测方法和缓解策略。

## 偏见的来源

### 偏见产生链

```
训练数据 → 标注过程 → 算法设计 → 模型训练 → Agent行为 → 社会影响
   ↓          ↓          ↓          ↓          ↓          ↓
 采样偏差   标注者偏见  特征选择   优化目标   推理偏差   反馈循环
 历史偏见   文化偏见    代理变量   过拟合    上下文偏见  歧视放大
```

### 统计检测框架

```python
import numpy as np
from scipy import stats
from typing import Dict, List, Tuple


class BiasDetector:
    """
    系统性偏见检测器。

    提供多角度的偏见检测方法。
    """

    def __init__(self, sensitive_attributes: List[str]):
        """
        sensitive_attributes: 受保护属性列表
        例如：["gender", "age_group", "ethnicity", "location"]
        """
        self.sensitive_attributes = sensitive_attributes

    def demographic_parity_test(self,
                                 outcomes: List[int],
                                 groups: List[str],
                                 privileged_group: str = None) -> dict:
        """
        人口平等性检测 (Demographic Parity)。

        检查不同群体获得正面结果的比例是否相同。

        Demographic Parity = P(positive | group=A) / P(positive | group=B)

        理想值：1.0（完全平等）
        业界标准：0.8 - 1.25（四分之五规则）
        """
        unique_groups = set(groups)
        group_outcomes = {}

        for group in unique_groups:
            group_indices = [i for i, g in enumerate(groups)
                           if g == group]
            group_positives = sum(outcomes[i] for i in group_indices)
            group_total = len(group_indices)
            group_outcomes[group] = {
                "positive_rate": group_positives / group_total
                if group_total > 0 else 0,
                "total": group_total
            }

        # 计算所有组对之间的比率
        disparities = {}
        group_list = list(unique_groups)
        for i in range(len(group_list)):
            for j in range(i + 1, len(group_list)):
                g1, g2 = group_list[i], group_list[j]
                rate1 = group_outcomes[g1]["positive_rate"]
                rate2 = group_outcomes[g2]["positive_rate"]
                if rate2 > 0:
                    ratio = rate1 / rate2
                    disparities[f"{g1}/{g2}"] = ratio

        # 检查是否违反四分之五规则
        violations = []
        for pair, ratio in disparities.items():
            if ratio < 0.8 or ratio > 1.25:
                violations.append({
                    "pair": pair,
                    "ratio": ratio,
                    "severity": "critical"
                    if ratio < 0.5 or ratio > 2.0
                    else "warning"
                })

        return {
            "group_statistics": group_outcomes,
            "disparity_ratios": disparities,
            "violations": violations,
            "has_bias": len(violations) > 0
        }

    def equalized_odds_test(self,
                             outcomes: List[int],
                             ground_truth: List[int],
                             groups: List[str]) -> dict:
        """
        均等化赔率检测 (Equalized Odds)。

        比Demographic Parity更严格：要求不同群体的
        真阳性率(TPR)和假阳性率(FPR)都相等。
        """
        unique_groups = set(groups)
        group_metrics = {}

        for group in unique_groups:
            group_indices = [i for i, g in enumerate(groups)
                           if g == group]

            tp = sum(1 for i in group_indices
                    if outcomes[i] == 1 and ground_truth[i] == 1)
            fp = sum(1 for i in group_indices
                    if outcomes[i] == 1 and ground_truth[i] == 0)
            tn = sum(1 for i in group_indices
                    if outcomes[i] == 0 and ground_truth[i] == 0)
            fn = sum(1 for i in group_indices
                    if outcomes[i] == 0 and ground_truth[i] == 1)

            tpr = tp / (tp + fn) if (tp + fn) > 0 else 0
            fpr = fp / (fp + tn) if (fp + tn) > 0 else 0

            group_metrics[group] = {"TPR": tpr, "FPR": fpr}

        # 计算TPR和FPR的最大差异
        tpr_values = [m["TPR"] for m in group_metrics.values()]
        fpr_values = [m["FPR"] for m in group_metrics.values()]

        max_tpr_diff = max(tpr_values) - min(tpr_values)
        max_fpr_diff = max(fpr_values) - min(fpr_values)

        return {
            "group_metrics": group_metrics,
            "max_tpr_difference": max_tpr_diff,
            "max_fpr_difference": max_fpr_diff,
            "has_bias": max_tpr_diff > 0.1 or max_fpr_diff > 0.1
        }

    def subgroup_analysis(self,
                           performance_metric: callable,
                           data: List[dict],
                           group_by: str) -> dict:
        """
        子群体分析：按不同维度分解性能指标。

        发现"平均表现好但某些子群体表现差"的问题。
        """
        subgroups = {}
        for item in data:
            group_val = item.get(group_by, "unknown")
            if group_val not in subgroups:
                subgroups[group_val] = []
            subgroups[group_val].append(item)

        results = {}
        for group, items in subgroups.items():
            score = performance_metric(items)
            results[group] = {
                "score": score,
                "sample_size": len(items)
            }

        # 识别表现异常的群体
        scores = [r["score"] for r in results.values()]
        mean_score = np.mean(scores)
        std_score = np.std(scores)

        anomalies = {}
        for group, info in results.items():
            z_score = (info["score"] - mean_score) / std_score \
                      if std_score > 0 else 0
            if abs(z_score) > 2:
                anomalies[group] = {
                    "score": info["score"],
                    "z_score": z_score,
                    "direction": "underperforming"
                    if z_score < 0 else "overperforming"
                }

        return {
            "subgroup_scores": results,
            "mean_score": mean_score,
            "std_score": std_score,
            "anomalies": anomalies,
            "worst_group": min(results.items(),
                             key=lambda x: x[1]["score"]),
            "best_group": max(results.items(),
                            key=lambda x: x[1]["score"]),
        }

    def counterfactual_fairness_test(self,
                                      agent,
                                      base_input: str,
                                      variations: Dict[str, str]) -> dict:
        """
        反事实公平性检测。

        核心思想：在改变敏感属性的同时保持其他条件不变，
        检测Agent的决策是否变化。

        例如：
        base: "一个经验丰富的工程师申请管理岗位"
        variation: "一个经验丰富的女工程师申请管理岗位"

        如果Agent对两者的评估显著不同，则存在偏见。
        """
        base_output = agent.decide(base_input)

        results = {}
        for attr_name, varied_input in variations.items():
            varied_output = agent.decide(varied_input)

            # 计算输出差异
            difference = self._compute_output_difference(
                base_output, varied_output
            )

            results[attr_name] = {
                "base_output": base_output,
                "varied_output": varied_output,
                "difference": difference,
                "is_biased": difference > 0.1  # 阈值可调
            }

        return {
            "base_input": base_input,
            "results": results,
            "has_bias": any(r["is_biased"] for r in results.values())
        }

    def _compute_output_difference(self, output_a, output_b) -> float:
        """计算两个输出的差异程度"""
        # 简化：如果是分类任务，检查标签是否一致
        if isinstance(output_a, str) and isinstance(output_b, str):
            return 0.0 if output_a == output_b else 1.0
        # 如果是连续值，计算相对差异
        if isinstance(output_a, (int, float)):
            return abs(output_a - output_b) / max(abs(output_a), 1e-6)
        return 0.0
```

```python
class DataLevelMitigation:
    """数据层面的偏见缓解"""

    @staticmethod
    def resample_for_balance(dataset: List[dict],
                              sensitive_attr: str) -> List[dict]:
        """
        重采样：平衡不同群体的样本量。

        策略选择：
        - Oversampling: 复制少数群体的样本
        - Undersampling: 减少多数群体的样本
        - SMOTE: 合成少数群体样本
        """
        from collections import Counter
        groups = Counter(item[sensitive_attr] for item in dataset)
        max_count = max(groups.values())

        balanced = []
        for group, count in groups.items():
            group_items = [item for item in dataset
                          if item[sensitive_attr] == group]
            if count < max_count:
                # Oversample少数群体
                multiplier = max_count // count
                remainder = max_count % count
                balanced.extend(group_items * multiplier)
                balanced.extend(
                    np.random.choice(group_items, remainder).tolist()
                )
            else:
                balanced.extend(group_items)

        return balanced

    @staticmethod
    def reweight_samples(dataset: List[dict],
                          sensitive_attr: str) -> List[dict]:
        """
        重加权：给少数群体更高的样本权重。

        相比重采样，不会改变数据集大小，但在训练时会
        给少数群体的样本更高的损失权重。
        """
        from collections import Counter
        groups = Counter(item[sensitive_attr] for item in dataset)
        n_total = len(dataset)
        n_groups = len(groups)

        for item in dataset:
            group = item[sensitive_attr]
            # 权重 = 总样本数 / (组数 * 该组样本数)
            item["sample_weight"] = n_total / (n_groups * groups[group])

        return dataset

    @staticmethod
    def data_augmentation_for_diversity(
            dataset: List[dict],
            underrepresented_groups: List[str]) -> List[dict]:
        """
        数据增强：为欠表示群体生成更多训练样本。

        使用LLM或规则生成具有多样性的合成数据。
        """
        augmented = list(dataset)

        for group in underrepresented_groups:
            # 使用LLM生成该群体的合成样本
            # 注意：必须保证合成数据的质量和真实性
            synthetic_samples = DataLevelMitigation._generate_synthetic(
                dataset, group, count=100
            )
            augmented.extend(synthetic_samples)

        return augmented
```

```python
class BiasPropagationAnalyzer:
    """
    多Agent系统中偏见传播的分析工具。

    在一个多Agent系统中，偏见可以通过以下路径传播：

    1. 串行传播：Agent A的偏见输出 → Agent B的输入 → Agent B的偏见输出
    2. 聚合放大：多个Agent的偏见在汇总时被放大
    3. 反馈循环：系统输出影响环境，环境变化进一步强化偏见
    """

    def __init__(self):
        self.agent_graph = {}  # agent_id -> set(downstream_agents)

    def add_dependency(self, upstream: str,
                        downstream: str):
        """记录Agent之间的依赖关系"""
        if upstream not in self.agent_graph:
            self.agent_graph[upstream] = set()
        self.agent_graph[upstream].add(downstream)

    def trace_bias_propagation(self,
                                source_agent: str,
                                bias_metric: str = "demographic_parity") -> dict:
        """
        追踪偏见从源头Agent向下游的传播路径。

        使用BFS找到所有可能受影响的Agent。
        """
        visited = set()
        propagation_path = []
        queue = [(source_agent, 0)]  # (agent_id, distance)

        while queue:
            current, distance = queue.pop(0)
            if current in visited:
                continue
            visited.add(current)
            propagation_path.append({
                "agent": current,
                "distance_from_source": distance,
                "risk_level": "critical" if distance <= 1
                         else "high" if distance <= 2
                         else "medium"
            })

            downstream = self.agent_graph.get(current, set())
            for ds in downstream:
                if ds not in visited:
                    queue.append((ds, distance + 1))

        return {
            "source_agent": source_agent,
            "affected_agents": len(visited) - 1,
            "propagation_path": propagation_path,
            "recommendation": (
                "在源头Agent处进行偏见检测和缓解，"
                "可以阻止偏见向下游传播"
            )
        }

    def simulate_bias_scenario(self,
                                initial_bias: float,
                                amplification_factor: float,
                                num_steps: int = 5) -> List[float]:
        """
        模拟偏见在多Agent系统中的累积放大效应。

        initial_bias: 初始偏见程度
        amplification_factor: 每一步的放大因子（>1放大，<1衰减）
        """
        bias_levels = [initial_bias]
        for step in range(1, num_steps):
            new_level = bias_levels[-1] * amplification_factor
            bias_levels.append(new_level)

        return bias_levels
```

## 小结

算法偏见不是技术问题，而是社会问题在技术系统中的映射。AI Agent不会自动变得公平——公平性需要通过刻意的设计、持续的监控和诚实的反思来实现。

**核心认知**：

1. **偏见无处不在**：不存在"无偏见"的AI系统，只存在"管理好偏见"的系统
2. **早发现早处理**：在数据和处理管道的前端解决偏见，比事后修补有效得多
3. **多维度检测**：Demographic Parity、Equalized Odds、Counterfactual Fairness等多指标互补
4. **系统化缓解**：数据重采样 + 算法约束 + 后处理校准的多层防御
5. **持续监控**：偏见不是一次性修复的，需要贯穿系统生命周期的持续关注

**对多Agent系统的特殊考量**：
- 偏见可以在Agent间传播和放大
- 在Agent通信接口处进行偏见检测
- 定期对整个Agent生态系统进行公平性评估

记住：**公平性不是终点，而是一个持续的、需要不断参与的过程**。构建公平的AI Agent系统，既是对用户的道德责任，也是系统长期可持续发展的基础。
