```python
def fuse_multimodal_inputs(inputs):
    """
    融合多模态输入为统一语义表示。
    """
    parts = ["[融合感知]"]
    # TODO: 遍历 inputs，拼接各模态信息
    for modal in ["text", "image_desc", "audio_transcript"]:
        if modal in inputs and inputs[modal]:

    return " ".join(parts)
```

编写类 `KnowledgeGraph` 和函数 `query_related_concepts(kg, entity, max_depth=1)`：

```python
class KnowledgeGraph:
    def __init__(self):
        self.graph = {}  # {entity: [(relation, target), ...]}

    def add_triple(self, subject, relation, obj):
        """添加 (subject, relation, object) 三元组"""
        if subject not in self.graph:
            self.graph[subject] = []
        self.graph[subject].append((relation, obj))

def query_related_concepts(kg, entity, max_depth=1):
    """
    BFS 从 entity 出发，返回 max_depth 步内可达的所有实体。
    """
    # TODO: BFS遍历
    visited = set()
    current = {entity}
    for _ in range(max_depth):
        next_level = set()
        for e in current:
            if e not in visited and e in kg.graph:
                visited.add(e)
                for rel, target in kg.graph[e]:
                    next_level.add(target)
        current = next_level
    visited.update(current)
    visited.discard(entity)
    return visited
```

```python
def decompose_task(goal, subtask_templates):
    """
    根据目标中的关键词匹配子任务模板，返回分解结果。
    """
    # TODO: 关键词匹配收集子任务
    tasks = []
    seen = set()
    for keyword, subtasks in subtask_templates.items():
        if keyword in goal:
            for t in subtasks:
                if t not in seen:
                    tasks.append(t)
                    seen.add(t)
    return tasks
```

```python
def parse_function_call(llm_response):
    """
    解析 LLM 的函数调用 JSON 并模拟执行。
    """
    func_name = llm_response.get("function", "")
    params = llm_response.get("params", {})
    result = None
    # TODO: 根据 func_name 调用对应模拟函数
    if func_name == "get_weather":
        result = f"{params.get('city','未知')}: 晴, 25°C"
    elif func_name == "calculate":
        try:
            result = eval(str(params.get("expr", "0")))
        except:
            result = "计算错误"
    elif func_name == "search":
        result = f"搜索结果: 关于'{params.get('query','')}'的3条信息"

        result = "未知函数"
    return {"function": func_name, "params": params, "result": result}
```

关卡 3-12：沙箱安全执行代码
所属模块：模块三 · 智能体四大核心能力模块 — 行动模块
--------------------------------------------------------------------------------

【任务描述】
智能体需要在受限的沙箱环境中安全执行代码，防止恶意代码访问文件系统或网络。

编写函数 `sandbox_execute(code, allowed_builtins=None)`：
- 仅允许使用白名单中的内置函数（默认：print, len, range, int, float, str, list, dict, abs, min, max, sum）
- 禁止 __import__、open、eval、exec 等危险函数
- 用受限的 globals 字典执行代码
- 返回执行结果字符串（captured stdout）或错误信息

【输入输出说明】
输入：code(str), allowed_builtins(set or None)
输出：{"success": bool, "output": str}

【初始代码】
```python
import io
import sys

def sandbox_execute(code, allowed_builtins=None):
    """
    在受限环境中执行 Python 代码。
    """
    if allowed_builtins is None:
        allowed_builtins = {"print", "len", "range", "int", "float", "str", "list", "dict", "abs", "min", "max", "sum"}
    safe_globals = {"__builtins__": {k: __builtins__[k] for k in allowed_builtins if k in __builtins__}}
    safe_globals["__builtins__"]["__import__"] = None  # 禁止import
    old_stdout = sys.stdout
    sys.stdout = io.StringIO()
    try:
        exec(code, safe_globals)
        output = sys.stdout.getvalue()
        return {"success": True, "output": output}
    except Exception as e:
        return {"success": False, "output": str(e)}
    finally:
        sys.stdout = old_stdout
```

【评测代码】
```python
def evaluate_3_12(student_func):
    r1 = student_func("print('hello sandbox')")
    assert r1["success"] and "hello sandbox" in r1["output"]
    r2 = student_func("print(len([1,2,3]))")
    assert r2["success"] and "3" in r2["output"]
    r3 = student_func("open('/etc/passwd')")
    assert not r3["success"], "open应被禁止"
    return True, "关卡通过：正确实现沙箱执行"
```

关卡 3-13：工具链编排设计
所属模块：模块三 · 智能体四大核心能力模块 — 行动模块
--------------------------------------------------------------------------------

【任务描述】
复杂任务需要多个工具按顺序协作：工具A的输出是工具B的输入，形成工具链。需要
处理工具间的依赖关系和错误传递。

编写函数 `execute_tool_chain(tool_chain, initial_input)`：
- tool_chain：[{name, func, error_handling}]
- 顺序执行每个工具，前一个的输出作为后一个的输入
- 如果某工具设置了 error_handling="skip"，出错时跳过并保持输入不变
- 返回最终结果和每步的执行状态

【输入输出说明】
输入：tool_chain(list[dict]), initial_input(any)
输出：{"final_result": any, "steps": [{"tool":..., "ok":..., "output":...}, ...]}

【初始代码】
```python
def execute_tool_chain(tool_chain, initial_input):
    """
    顺序执行工具链，支持错误跳过。
    """
    current = initial_input
    steps = []
    for tc in tool_chain:
        try:
            current = tc["func"](current)
            steps.append({"tool": tc["name"], "ok": True, "output": str(current)})
        except Exception as e:
            if tc.get("error_handling") == "skip":
                steps.append({"tool": tc["name"], "ok": False, "output": str(e)})
            else:
                return {"final_result": None, "steps": steps, "error": str(e)}
    return {"final_result": current, "steps": steps}
```

【评测代码】
```python

    chain = [
        {"name": "double", "func": lambda x: x * 2, "error_handling": "stop"},
        {"name": "add10", "func": lambda x: x + 10, "error_handling": "skip"},
        {"name": "to_str", "func": lambda x: str(x), "error_handling": "stop"},
    ]
    r = student_func(chain, 5)
    assert r["final_result"] == "20", f"5→10→20→'20', 得到{r['final_result']}"
    assert len(r["steps"]) == 3 and all(s["ok"] for s in r["steps"])
    return True, "关卡通过：正确实现工具链编排"
```
